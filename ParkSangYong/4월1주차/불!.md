import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.InputStreamReader;
import java.io.OutputStreamWriter;
import java.util.Arrays;
import java.util.Comparator;
import java.util.PriorityQueue;
import java.util.StringTokenizer;
import java.util.*;
 
public class Main {

    static int R, C;
    static int[][] arr;
    static Queue<Pair> queue = new LinkedList<>();
    static Queue<Pair> fire = new LinkedList<>();
 
    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        StringTokenizer st = new StringTokenizer(br.readLine());
        StringBuilder sb = new StringBuilder();
        R = Integer.parseInt(st.nextToken());
        C = Integer.parseInt(st.nextToken());

        arr = new int[R][C];

        for(int i = 0; i<R; i++){
            String line = br.readLine();
            for(int j = 0; j<C; j++){
                char chr = line.charAt(j);
                if(chr =='#') arr[i][j] = -1;
                else if(chr =='F') {
                    arr[i][j] = -2;
                    fire.add(new Pair(i, j, 0));
                }
                else if(chr =='J') {
                    arr[i][j] = 1;
                    queue.add(new Pair(i, j, 0));
                }
            }
        }
        bfs();
    }

    public static void bfs() {

        int[] x = {-1, 0, 1, 0};
        int[] y = {0, -1, 0, 1};

        int answer = 0;
        while(true){
            answer++;

            int fs  = fire.size();
            while(fs>0){
                fs--;
                Pair fTemp = fire.poll();
                int b = fTemp.b;
                int a = fTemp.a;
                for(int i=0; i<4; i++){
                    if (a+x[i] >= 0 && a+x[i] < R && b+y[i]>= 0 && b+y[i] < C && arr[a+x[i]][b+y[i]] >= 0){
                        fire.add(new Pair(a+x[i], b+y[i], fTemp.cnt+1));
                        arr[a+x[i]][b+y[i]] = -2;
                    }
                }
            }

            int qs = queue.size();
            while(qs > 0) {
                qs--;
                Pair temp = queue.poll();

                if(temp.a==R-1 || temp.b==C-1 || temp.a == 0 || temp.b == 0) {
                    System.out.println(answer);
                    return;
                }

                for(int i = 0; i<4; i++){
                    if(arr[temp.a+x[i]][temp.b+y[i]] == 0) {
                        Pair p = new Pair(temp.a+x[i], temp.b+y[i], temp.cnt+1);
                        arr[temp.a+x[i]][temp.b+y[i]] = 1;
                        queue.add(p);
                    }
                }
            }
            if(queue.isEmpty()){
                System.out.println("IMPOSSIBLE");
                return;
            }
        }
    }

    public static class Pair {
        int a;
        int b;
        int cnt;

        public Pair(int a, int b, int cnt) {
            this.a = a;
            this.b = b;
            this.cnt = cnt;
        }
        
        @Override
        public boolean equals(Object o) {
            Pair p = (Pair) o;

            return (p.a == this.a && p.b == this.b);
        }

        @Override
        public int hashCode() {
            return (""+a+b).hashCode();
        }
    }
 
}
